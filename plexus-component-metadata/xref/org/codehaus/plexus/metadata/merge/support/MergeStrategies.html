27  import java.util.Collections;
28  import java.util.List;
29  
30  import org.codehaus.plexus.metadata.merge.MergeException;
31  import org.codehaus.plexus.metadata.merge.MergeStrategy;
32  
33  /**
34   * Collection of available Merge Strategies.<p>
35   * TODO: Revisit and factor {@link Mergeable#merge(Mergeable)} to use a {@link MergeStrategy}.
36   *
37   * @author <a href='mailto:rahul.thakur.xdev@gmail.com'>Rahul Thakur</a>
38   */
39  public class MergeStrategies
40  {
41      /**
42       * {@link MergeStrategy} implementation wherein the elements are merged
43       * down to the deepest available {@link Mergeable} instance in the DOM tree.
44       */
45      public static final MergeStrategy DEEP = new MergeStrategy()
46      {
47          public void apply( Mergeable dElt, Mergeable rElt )
48              throws MergeException
49          {
50              dElt.merge( rElt );
51          }
52      };
53  
54      /**
55       * {@link MergeStrategy} implementation wherein only the element on
56       * which the merge operation is called is 'merged'. The merge does not
57       * traverse the DOM tree any further.
58       */
59      public static final MergeStrategy SHALLOW = new MergeStrategy()
60      {
61          /**
62           * @throws MergeException
63           * @see org.codehaus.plexus.metadata.merge.MergeStrategy#apply(Mergeable,Mergeable)
64           */
65          public void apply( Mergeable dElt, Mergeable rElt )
66              throws MergeException
67          {
68              AbstractMergeableElement dame = (AbstractMergeableElement) dElt;
69              AbstractMergeableElement rame = (AbstractMergeableElement) rElt;
70  
71              // check if the dominant was in conflict with recessive.
72              List elementNames = dame.getElementNamesForConflictResolution( Collections.EMPTY_LIST );
73  
74              if ( !dame.isRecessiveElementInConflict( rame, elementNames ) )
75              {
76                  // no conflict, simply add recessive to dominant's parent
77                  dame.getElement().addContent( rame.getElement() );
78              }
79          }
80      };
81  }
